5. Make a TABLE

# Reshape
tx_wide <- tx %>%
  select(NAME, variable, estimate, moe, geometry) %>%
  pivot_wider(
    id_cols = c(NAME, geometry),
    names_from = variable,
    values_from = c(estimate, moe),
    names_glue = "{variable}_{.value}" )

names(tx_wide)
[1] "NAME"                   "geometry"               "male_total_estimate"   
[4] "female_total_estimate"  "median_income_estimate" "male_total_moe"        
[7] "female_total_moe"       "median_income_moe"     
# Male / Female Ratio
tx_wide <- tx_wide %>%
  mutate(male_female_ratio = male_total_estimate / female_total_estimate)

View(tx_wide)


6. Make MAPs

# Choropleth of Median Household Income
ggplot(tx_wide) +
  geom_sf(aes(fill = median_income_estimate), color = "white", size = 0.1) +
  scale_fill_viridis_c(
    option = "turbo",
    name = "Median Income ($)",
    labels = scales::dollar_format()  ) +
  labs(
    title = "Median Household Income by County (Texas, 2023)",
    caption = "Source: U.S. Census Bureau, ACS 5-Year Estimates"  ) +
  theme_minimal()
# Choropleth of Male/Female Ratio
ggplot(tx_wide) +
  geom_sf(aes(fill = male_female_ratio), color = "white", size = 0.1) +
  scale_fill_viridis_c(
    option = "plasma",
    name = "Male/Female Ratio",
    direction = -1  ) +
  labs(
    title = "Male-to-Female Ratio by County (Texas, 2023)",
    caption = "Source: U.S. Census Bureau, ACS 5-Year Estimates"  ) +
  theme_minimal()


6. INTERPRETATION

The choropleth map highlights income disparities across Texas. While urban counties exhibit higher median household incomes, rural and border counties tend to have lower income levels. The gender ratio table reveals notable demographic differences. In some rural counties, the male population exceeds the female population, which may be associated with industry-specific employment patterns or migration trends.
